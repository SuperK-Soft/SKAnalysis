200 bool DataModel::AddCut(std::string selector, std::string cut, std::string description, bool savedist, Args... rest){
201  if(selector!="all" && Selectors.count(selector)==0){
202  std::cerr<<"DataModel::AddCut Error! Unrecognised selector "<<selector
203  <<" for cut "<<cut<<std::endl;
204  return false;
205  }
206  bool ret=true;
207  if(selector=="all"){
208  // add this event to all selectors
209  for(auto sel = Selectors.begin(); sel!=Selectors.end(); ++sel){
210  ret &= sel->second->AddCut(cut, description, savedist, rest...);
211  }
212  } else {
213  ret = Selectors.at(selector)->AddCut(cut, description, savedist, rest...);
214  }
215  return ret;

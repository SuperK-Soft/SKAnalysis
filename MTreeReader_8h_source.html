98  // (the user really ought not to pass us an array reference, as it requires

101  template<typename T>
102  int GetArrayBranchValue(std::string branchname, T* arr_in, std::size_t NCOL, std::size_t NROW=1, std::size_t NAISLE=1){
103  // check we know this branch
104  if(branch_value_pointers.count(branchname)==0){
105  std::cerr<<"No such branch "<<branchname<<std::endl;
106  return 0;
107  }
108  // check if the branch is an array - this template specialization is only for arrays
109  if(not branch_isarray.at(branchname)){
110  std::cerr<<"Branch "<<branchname
111  <<" is not an array; please check your datatype to GetBranchValue()"<<std::endl;
112  return 0;
113  }
114  // check the passed array has suitable dimensions.
115  // first we need to know the actual array dimensions
116  std::vector<size_t> branchdims = GetBranchDims(branchname);
117  // for dynamic arrays we may need to update our pointer to the stored array
118  // not sure if we should bail if the user is trying to put a dynamic array
119  // into a static-sized array variable.... continue for now.
120  UpdateBranchPointer(branchname);
121 
122  // the user's array must be at least as large as required
123  // first check the number of dimensions is sufficient
124  int ndims = 1;
125  if(NROW>1) ++ndims;
126  if(NAISLE>1) ++ndims;
127  if(ndims!=branchdims.size()){
128  std::cerr<<"passed an array reference of dimensionality "<<ndims
129  <<" for branch "<<branchname<<" which has dimensionality "
130  <<branchdims.size()<<std::endl;
131  return 0;
132  }
133 
134  // next check each dimension has sufficient capacity
135  bool cap_ok = true;
136  if(NCOL<branchdims.at(0)) cap_ok = false;
137  if(ndims>1 && NROW<branchdims.at(1)) cap_ok = false;
138  if(ndims>2 && NAISLE<branchdims.at(2)) cap_ok = false;
139  if(not cap_ok){
140  std::cerr<<"passed an array with dimensions ["<<NCOL<<"]";
141  if(NROW>1) std::cerr<<"["<<NROW<<"]";
142  if(NAISLE>1) std::cerr<<"["<<NAISLE<<"] ";
143  std::cerr<<" which is insufficient for the data dimensions ";
144  for(int i=0; i<ndims; ++i) std::cerr<<"["<<branchdims.at(i)<<"]";
145  std::cerr<<std::endl;
146  return 0;
147  }
148 
149  // copy the data to the user's array
150  int data_cols = branchdims.at(0);
151  int data_rows = (ndims>1) ? branchdims.at(1) : 1;
152  int data_aisles = (ndims>2) ? branchdims.at(2) : 1;
153  T* objp = reinterpret_cast<T*>(branch_value_pointers.at(branchname));
154  for(int aisle=0; aisle<NAISLE; ++aisle){
155  for(int row=0; row<NROW; ++row){
156  for(int col=0; col<NCOL; ++col){
157  std::size_t dest_flat_index = aisle*NCOL*NROW + row*NCOL + col;
158  std::size_t source_flat_index = aisle*data_rows*data_cols + row*data_cols + col;
159  if((col+1)>branchdims.at(0) ||
160  (ndims>1 && (row+1)>branchdims.at(1)) ||
161  (ndims>2 && (aisle+1)>branchdims.at(2))){
162  arr_in[dest_flat_index] = 0;
163  } else {
164  arr_in[dest_flat_index] = objp[source_flat_index];
165  }
166  }
167  }
168  }
169  return 1;
170  }
